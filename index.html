// ============================================================ // BATTLE.JS — НОВАЯ БОЕВАЯ СИСТЕМА // ============================================================ class Battle { constructor(player, enemy) { this.player = player; // { hp, maxHp, damage, defense, agility, army } this.enemy = enemy; // { hp, maxHp, damage, defense, agility, army, morale } this.turn = 'player'; this.phase = 'active'; // 'active' | 'victory' | 'defeat' this.log = []; this.timer = null; this.strategy = 'attack'; // 'attack' | 'defend' | 'flank' | 'siege' this.turnTime = 10000; // 10 секунд на ход } // ============================================================ // СТАРТ БОЯ // ============================================================ start() { this.log.push('⚔️ Бой начался!'); this.nextTurn(); return this; } // ============================================================ // ХОД // ============================================================ nextTurn() { if (this.phase !== 'active') return; // Проверка победы/поражения if (this.enemy.army <= 0 || this.enemy.hp <= 0) { this.phase = 'victory'; this.log.push('🎉 ПОБЕДА! Враг разбит!'); this.stop(); return; } if (this.player.army <= 0 || this.player.hp <= 0) { this.phase = 'defeat'; this.log.push('💀 ПОРАЖЕНИЕ... Ваша армия уничтожена.'); this.stop(); return; } // Ход игрока if (this.turn === 'player') { this.log.push(`⏳ Ваш ход. Выберите стратегию: ${this.strategy}`); // Ждём выбора игрока (через интерфейс) } else { // Ход врага (ИИ) this.enemyTurn(); } } // ============================================================ // ВЫБОР СТРАТЕГИИ (игрок) // ============================================================ setStrategy(strategy) { if (this.turn !== 'player') return; this.strategy = strategy; this.log.push(`📯 Стратегия выбрана: ${strategy}`); this.playerTurn(); } // ============================================================ // ХОД ИГРОКА // ============================================================ playerTurn() { const p = this.player; const e = this.enemy; let damage = 0; let message = ''; // Расчёт урона в зависимости от стратегии switch (this.strategy) { case 'attack': damage = (p.damage + p.army * 0.2) * (0.8 + Math.random() * 0.4); e.army -= damage; message = `⚔️ Атака! Нанесено ${Math.round(damage)} урона.`; break; case 'defend': const heal = p.army * 0.05 + 10; p.army += heal; message = `🛡️ Защита! Восстановлено ${Math.round(heal)} солдат.`; break; case 'flank': damage = (p.damage + p.agility * 0.5) * (0.6 + Math.random() * 0.6); e.army -= damage; e.morale -= 5; message = `🏹 Фланговая атака! Нанесено ${Math.round(damage)} урона. Мораль врага -5.`; break; case 'siege': damage = (p.damage * 1.5) * (0.5 + Math.random() * 0.5); e.hp -= damage; message = `🔥 Осадные орудия! Нанесено ${Math.round(damage)} урона стенам.`; break; default: damage = p.damage * 0.5; e.army -= damage; message = `⚔️ Атака! Нанесено ${Math.round(damage)} урона.`; } // Ответ врага const enemyResponse = this.enemyResponse(); message += '\n' + enemyResponse; this.log.push(message); this.turn = 'enemy'; // Таймер на ход врага this.timer = setTimeout(() => { this.nextTurn(); }, this.turnTime); } // ============================================================ // ХОД ВРАГА (ИИ) // ============================================================ enemyTurn() { const p = this.player; const e = this.enemy; // ИИ выбирает стратегию в зависимости от состояния let enemyStrategy = 'attack'; if (e.morale < 30) enemyStrategy = 'defend'; else if (p.army < e.army * 0.5) enemyStrategy = 'attack'; else if (Math.random() < 0.3) enemyStrategy = 'flank'; let damage = 0; let message = ''; switch (enemyStrategy) { case 'attack': damage = (e.damage + e.army * 0.15) * (0.7 + Math.random() * 0.5); p.army -= damage; message = `🐺 Враг атакует! Потеряно ${Math.round(damage)} солдат.`; break; case 'defend': const heal = e.army * 0.03 + 5; e.army += heal; message = `🛡️ Враг защищается! Восстановил ${Math.round(heal)} солдат.`; break; case 'flank': damage = (e.damage + e.agility * 0.3) * (0.5 + Math.random() * 0.6); p.army -= damage; message = `🏹 Враг обходит фланг! Потеряно ${Math.round(damage)} солдат.`; break; } this.log.push(message); this.turn = 'player'; // Таймер на ход игрока this.timer = setTimeout(() => { this.nextTurn(); }, this.turnTime); } // ============================================================ // ОТВЕТ ВРАГА НА ДЕЙСТВИЯ ИГРОКА // ============================================================ enemyResponse() { const p = this.player; const e = this.enemy; let damage = 0; let message = ''; // Враг отвечает на атаку if (this.strategy === 'attack' || this.strategy === 'flank') { damage = (e.damage * 0.5 + e.army * 0.05) * (0.5 + Math.random() * 0.5); p.army -= damage; message = `🏹 Враг отвечает! Потеряно ${Math.round(damage)} солдат.`; } else if (this.strategy === 'siege') { // Если игрок использует осадные орудия, враг пытается их уничтожить const siegeDamage = e.army * 0.02 + 5; p.army -= siegeDamage; message = `🔥 Враг атакует осадные орудия! Потеряно ${Math.round(siegeDamage)} солдат.`; } else { damage = e.damage * 0.2; p.army -= damage; message = `💢 Враг контратакует! Потеряно ${Math.round(damage)} солдат.`; } // Снижение морали врага e.morale = Math.max(0, e.morale - 0.5); return message; } // ============================================================ // ОСТАНОВКА БОЯ // ============================================================ stop() { if (this.timer) { clearTimeout(this.timer); this.timer = null; } this.phase = this.phase === 'active' ? 'stop' : this.phase; } // ============================================================ // ПОЛУЧИТЬ ЛОГ // ============================================================ getLog() { return this.log.slice(-10); } // ============================================================ // ПОЛУЧИТЬ СТАТУС // ============================================================ getStatus() { return { player: { hp: Math.round(this.player.hp), army: Math.round(this.player.army), maxArmy: this.player.maxArmy || 100 }, enemy: { hp: Math.round(this.enemy.hp), army: Math.round(this.enemy.army), morale: Math.round(this.enemy.morale) }, phase: this.phase, turn: this.turn, strategy: this.strategy, log: this.getLog() }; } } // ============================================================ // СОЗДАНИЕ БОЯ (ИСПОЛЬЗОВАНИЕ) // ============================================================ function createBattle(playerStats, enemyStats) { const battle = new Battle(playerStats, enemyStats); return battle.start(); } // ============================================================ // ПРИМЕР ИСПОЛЬЗОВАНИЯ (для теста) // ============================================================ /* const player = { hp: 100, maxHp: 100, damage: 20, defense: 10, agility: 15, army: 100, maxArmy: 100 }; const enemy = { hp: 80, maxHp: 80, damage: 15, defense: 8, agility: 10, army: 80, morale: 100 }; const battle = createBattle(player, enemy); // ... и далее через интерфейс вызываем battle.setStrategy('attack') и т.д. */
